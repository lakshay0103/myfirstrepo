class Solution{
    public:
    
    pair<bool,int> find(Node * root){
        
        if(root==NULL){
            pair<bool,int> p=make_pair(true,0);
            return p;
        }
        
        pair<bool,int> left=find(root->left);
        pair<bool,int> right=find(root->right);
        
        bool leftans=left.first;
        bool rightans=right.first;
        
        bool diff=abs(left.second-right.second)<=1;
        
        pair<bool,int> ans;
        ans.second=max(left.second,right.second)+1;
        
        if(leftans and rightans and diff){
            ans.first=true;
        }else
        {
            ans.first=false;
        }
        
        return ans;
    }
    //Function to check whether a binary tree is balanced or not.
    bool isBalanced(Node *root)
    {
        return find(root).first;
    }
};

class Solution{

    public:

    //Function to check whether a binary tree is balanced or not.

    

    pair<int,bool> bal(Node* root)

    {

        

        if(root == NULL)

        {

            pair<int,bool> temp = make_pair(0,true);

            return temp;

        }

        

        pair<int,bool> left = bal(root->left);

        pair<int,bool> right = bal(root->right);

        

        //this chunk of code is to test the height condition 

        int heightCheck =  abs(left.first - right.first);

        bool con1;

        if(heightCheck < 2)

        {

            con1 = true;

        }

        else{

            con1 = false;

        }

        

        

        //this chunk of code id to test the balanced subtree condition

        bool con2;

        if(left.second && right.second)

        {

            con2 = true;

        }

        else

        {

            con2 = false;

        }

        

        bool condition;

            condition = true;

        }

        else{

            condition = false;

        }

        

        

        //this chunk of code is to update the height

        int height = max(left.first,right.first) + 1;

        

       

     //updating the pair

         pair<int,bool> answer;

        answer.first = height;

        answer.second = condition;

             return answer; 

    }
